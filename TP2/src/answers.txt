-------> ler um inteiro N, depois ler N numeros e escrever o menor deles.
STARTDECL
int minimo, n, num, min;
ENDDECL
STARTBODY
read(n);
read(min);
repeat{
    read(num);
    if(min > num){
        min = num;
    }
    n = n - 1; 
}until(n>1)             
print (minimo);
ENDBODY
